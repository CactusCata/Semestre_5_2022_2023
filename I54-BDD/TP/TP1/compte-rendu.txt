CREATE TABLE Departement
(
  DID VARCHAR(2) PRIMARY KEY,
  Libelle VARCHAR(20)
);

CREATE TABLE Employe
(
  EID INT CHECK (EID < 100),
  Nom VARCHAR(20),
  Dept VARCHAR(2),

  PRIMARY KEY(EID),
  FOREIGN KEY(Dept) REFERENCES Departement(DID)
);
